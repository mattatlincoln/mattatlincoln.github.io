    Reading Files¶

    Be careful where you place you files and project

  • Do not place it on a removable disk or network drive
  • The easiest is to have your file in the same directory as your cpp file
  • Here is a minimal test case (break down the problem) that should read in a file containing a single integer
    • 
#include <iostream>
#include <stream>
#include <string>      
int main() {
  // read in problem array
  string nfile, temp;
  int N = 0;
  cout << "Insert the name of the file containing the coefficient and the constant terms" << "\n";
  cin >> nfile;
  ifstream myFile(nfile);
  if (myFile.fail())            //is it ok?
  {
    cout << "Input file did not open please check it\n";
    system("pause");
    return 1;
  }
  // Read the size of the matrix
  if (myFile.is_open()) {
    cout << "\nopened file\n\n";
    getline(myFile, temp, ',');
    N = stoi(temp);
    cout << "found file: N = " << N << "\n\n";
    getline(myFile, temp, '\n');
  }

  myFile.close();

}
